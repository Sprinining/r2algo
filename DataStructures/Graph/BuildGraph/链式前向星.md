## 链式前向星

链式前向星是一种使用静态链表实现邻接表的图存储结构。

优势在于用数组代替动态容器：

- **空间按需分配**：只为真实存在的边开空间 → O(n + m)，不会每个点都预留一大块数组。
- **内存连续**：所有边在一块数组里 → 缓存友好 → 常数更小。
- **无动态扩容**：不依赖 `vector push_back` → 性能稳定、可控。

```c
#include <stdio.h>
#include <string.h>

#define MAXN 1005      // 最大点数
#define MAXM 2005      // 最大边数（无向图记得 *2）

// 边结构体
struct Edge {
    int to;        // 终点
    int weight;    // 权重
    int next;      // 下一条边编号
};

struct Edge edges[MAXM];
int head[MAXN];    // head[u] = 第一条边编号
int cnt;           // 当前边编号

// 初始化图
void init() {
    cnt = 1;                   // 0 表示没有边
    memset(head, 0, sizeof(head));
}

// 加一条 u -> v 权重 w 的边
void addEdge(int u, int v, int w) {
    edges[cnt].to = v;
    edges[cnt].weight = w;
    edges[cnt].next = head[u];  // 插到原链表头前面
    head[u] = cnt;              // 更新头
    cnt++;
}

// 建有向图
void buildDirected(int edge[][3], int m) {
    for (int i = 0; i < m; i++) {
        addEdge(edge[i][0], edge[i][1], edge[i][2]);
    }
}

// 建无向图
void buildUndirected(int edge[][3], int m) {
    for (int i = 0; i < m; i++) {
        addEdge(edge[i][0], edge[i][1], edge[i][2]);
        addEdge(edge[i][1], edge[i][0], edge[i][2]);
    }
}

// 遍历图
void printGraph(int n) {
    for (int i = 1; i <= n; i++) {
        printf("%d: ", i);
        for (int ei = head[i]; ei != 0; ei = edges[ei].next) {
            printf("(%d,%d) ",
                   edges[ei].to,
                   edges[ei].weight);
        }
        printf("\n");
    }
}

int main() {
    int n = 4;
    int m = 4;

    // 每条边格式：{u, v, w}
    int input[4][3] = {
        {1, 2, 5},
        {1, 3, 2},
        {2, 4, 7},
        {3, 4, 6}
    };

    init();
    buildUndirected(input, m);   // 无向图
    printGraph(n);

    return 0;
}
```
